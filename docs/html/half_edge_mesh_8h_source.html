462  void computeCentroid(void) {
463  centroid = Vector3D( 0., 0., 0. ); // centroid
464  double d = 0.; // degree (i.e., number of neighbors)
465 
466  // Iterate over neighbors.
467  HalfedgeCIter h = halfedge();
468  do {
469  // Add the contribution of the neighbor,
470  // and increment the number of neighbors.
471  centroid += h->next()->vertex()->position;
472  d += 1.;
473 
474  h = h->twin()->next();
475  } while( h != halfedge() );
476 
477  centroid /= d; // compute the average
478  }
479 
484  Vector3D centroid;
485 
492  void computeNormal(void) {
493  normal = Vector3D( 0., 0., 0. );
494  Vector3D pi = position;
495 
496  // Iterate over neighbors.
497  HalfedgeCIter h = halfedge();
498  do {
499  Vector3D pj = h->next()->vertex()->position;
500  Vector3D pk = h->next()->next()->vertex()->position;
501 
502  // NOTE(kayvonf): This seems to be backward.
503  // Fixing to pj-pi cross pk-pi
504  //normal += cross( pk-pi, pj-pi );
505  normal += cross( pj-pi, pk-pi );
506 
507  h = h->twin()->next();
508  } while( h != halfedge() );
509 
510  normal.normalize();
511  }

529  do {
